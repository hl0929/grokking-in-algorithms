def min_window(s, t):
    if not s or not t: return ""
    need = {}
    for ch in t: need[ch] = need.get(ch, 0) + 1
    missing = len(t)
    left = start = end = 0
    for right in range(len(s)):
        ch = s[right]
        if need.get(ch, 0) > 0:
            missing -= 1
        need[ch] = need.get(ch, 0) - 1
        if missing == 0:
            while left < right and need[s[left]] < 0:
                need[s[left]] += 1
                left += 1
            if end == 0 or (right + 1 - left) < end - start:
                start, end = left, right + 1
    return s[start:end]
