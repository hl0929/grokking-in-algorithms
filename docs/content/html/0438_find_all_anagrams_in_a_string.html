def find_anagrams(s, p):
    if not s or not p or len(s) < len(p): return []
    need = [0] * 26
    for ch in p: need[ord(ch) - ord("a")] += 1
    need_count = sum(1 for i in need if i > 0)
    res, window, valid, left = [], [0] * 26, 0, 0
    for right, ch in enumerate(s):
        idx = ord(ch) - ord("a")
        window[idx] += 1
        if window[idx] == need[idx]: valid += 1
        if right - left + 1 > len(p):
            l_idx = ord(s[left]) - ord("a")
            if window[l_idx] == need[l_idx]: valid -= 1
            window[l_idx] -= 1
            left += 1
        if right - left + 1 == len(p) and valid == need_count:
            res.append(left)
    return res
