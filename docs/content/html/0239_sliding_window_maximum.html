def max_sliding_window(nums, k):
    if not nums or k == 0: return []
    from collections import deque
    dq = deque()
    res = []
    for idx, num in enumerate(nums):
        while dq and nums[dq[-1]] < num:
            dq.pop()
        dq.append(idx)
        # --- 观察下面这两行 ---
        if dq[0] <= idx - k:
            dq.popleft() # <--- 触发点
        if idx >= k - 1:
            res.append(nums[dq[0]])
    return res
