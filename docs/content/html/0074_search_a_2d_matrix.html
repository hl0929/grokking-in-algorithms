def search_matrix(matrix, target):
    m, n = len(matrix), len(matrix[0])
    left, right = 0, m * n - 1
    while left <= right:
        mid = (left + right) // 2
        row = mid // n
        col = mid % n
        val = matrix[row][col]
        if val == target:
            return True
        elif val > target:
            right = mid - 1
        else:
            left = mid + 1
    return False
