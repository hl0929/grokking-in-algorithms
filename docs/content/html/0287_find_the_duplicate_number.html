def find_duplicate(nums):
    slow = 0
    fast = 0
    while True: # 阶段 1: 寻找环内相遇点
        slow = nums[slow]
        fast = nums[nums[fast]]
        if slow == fast:
            break
    slow = 0 # 阶段 2: 寻找入口点
    while slow != fast:
        slow = nums[slow]
        fast = nums[fast]
    return slow
