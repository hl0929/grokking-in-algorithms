def is_palindrome(head):
# 1. 找中点 (快慢指针)
slow = fast = head
while fast and fast.next:
slow = slow.next
fast = fast.next.next
# 2. 反转后半部分
prev, curr = None, slow
while curr:
nxt = curr.next
curr.next = prev
prev = curr
curr = nxt
# 3. 比较左右两段
left, right = head, prev
while right:
if left.val != right.val:
return False
left = left.next
right = right.next
return True
